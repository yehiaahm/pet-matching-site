## 💻 الكود المثال

### استخدام بسيط
```tsx
import { mockPets } from '../data/mockPets';
import { PetShowcase } from './components/PetShowcase';
import { QuickAddPetButton } from './components/QuickAddPetButton';

export function SearchPage() {
  const [pets, setPets] = useState(mockPets);

  const handleAddPet = (newPet) => {
    setPets([...pets, newPet]);
  };

  return (
    <div>
      <h1>البحث عن حيوانك الأليف</h1>
      
      <PetShowcase 
        maxPets={12}
        onPetSelect={(pet) => console.log(pet)}
      />

      <QuickAddPetButton onAddPet={handleAddPet} />
    </div>
  );
}
```

export function AdminPage() {
